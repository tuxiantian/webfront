### 数组排序

```js
var arr = ["a", "b", "A", "B"];
arr.sort();
console.log(arr);

var arr = [23, 9, 4, 78, 3];
var compare = function (x, y) {//比较函数
    if (x < y) {
        return -1;
    } else if (x > y) {
        return 1;
    } else {
        return 0;
    }
}
console.log(arr.sort(compare));
```
